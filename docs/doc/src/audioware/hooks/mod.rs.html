73#[macro_export]
74macro_rules! attach_native_func {
75    ($name:literal, $offset:path, $hook: ident, $me:ident, $to:ident $(, $v:vis)?) => {
76        ::red4ext_rs::hooks! {
77            static $hook: fn(
78                i: *mut ::red4ext_rs::types::IScriptable,
79                f: *mut ::red4ext_rs::types::StackFrame,
80                a3: ::red4ext_rs::VoidPtr,
81                a4: ::red4ext_rs::VoidPtr) -> ();
82        }
83
84        #[allow(clippy::missing_transmute_annotations)]
85        $($v)? fn $me(env: &::red4ext_rs::SdkEnv) {
86            let addr = ::red4ext_rs::addr_hashes::resolve($offset);
87            let addr = unsafe { ::std::mem::transmute(addr) };
88            unsafe { env.attach_hook($hook, addr, $to) };
89            $crate::utils::intercept!("attached native func hook for {}", $name);
90        }
91    };
92    ($name:literal, $offset:path) => {
93        attach_native_func!($name, $offset, HOOK, attach_hook, detour, pub);
94    };
95}

97#[macro_export]
98macro_rules! attach_native_event {
99    ($offset:path, $class:path, $to:ident $(, $v:vis)?) => {
100        ::red4ext_rs::hooks! {
101            static HOOK: fn(
102                a1: *mut ::red4ext_rs::types::IScriptable,
103                a2: *mut $class) -> ();
104        }
105
106        #[allow(clippy::missing_transmute_annotations)]
107        $($v)? fn attach_hook(env: &::red4ext_rs::SdkEnv) {
108            let addr = ::red4ext_rs::addr_hashes::resolve($offset);
109            let addr = unsafe { ::std::mem::transmute(addr) };
110            unsafe { env.attach_hook(HOOK, addr, $to) };
111            $crate::utils::intercept!("attached native event hook for {}", <$class as ::red4ext_rs::ScriptClass>::NAME);
112        }
113    };
114    ($offset:path, $class:path) => {
115        attach_native_event!($offset, $class, detour, pub);
116    };
