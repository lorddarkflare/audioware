70    pub fn try_new(settings: AudioManagerSettings<B>) -> Result<Engine<B>, Error> {
71        let (banks, report) = Banks::new();
72        #[cfg(not(feature = "hot-reload"))]
73        let _ = BANKS.set(banks.clone());
74        #[cfg(feature = "hot-reload")]
75        {
76            *BANKS.write() = Some(banks.clone());
77        }
78        let mut manager = AudioManager::new(settings).map_err(|_| Error::Engine {
79            source: EngineError::Manager {
80                origin: "audio manager",
81            },
82        })?;
83        let modulators = Modulators::try_new(&mut manager)?;
84        let tracks = Tracks::try_new(&mut manager, &modulators)?;
85        Ok(Engine {
86            banks,
87            manager,
88            scene: None,
89            modulators,
90            tracks,
91            report,
92            last_volume: None,
93        })
94    }

104    pub fn report_initialization(&self, hot_reload: bool) {
105        let conjugation = if cfg!(feature = "hot-reload") && hot_reload {
106            "hot-reloaded"
107        } else {
108            "initialized"
109        };
110        let infinitive = if cfg!(feature = "hot-reload") && hot_reload {
111            "hot-reload"
112        } else {
113            "initialize"
114        };
115
116        match self.report.outcome() {
117            InitializationOutcome::Success => {
118                success!(["{}", self.report];["Audioware {} successfully!", conjugation]);
119            }
120            InitializationOutcome::PartialSuccess => {
121                warns!(["{}", self.report];["Audioware partially failed to {infinitive}. See RED4ext log for more details."]);
122            }
123            InitializationOutcome::CompleteFailure => {
124                fails!(["{}", self.report];["Audioware failed to {infinitive}. See RED4ext log for more details."]);
125            }
126        };
127    }

144    pub fn play_over_the_phone(
145        &mut self,
146        event_name: CName,
147        emitter_name: CName,
148        gender: audioware_manifest::PlayerGender,
149    ) {
150        let spoken = SpokenLocale::get();
151        match self.banks.try_get(&event_name, &spoken, Some(&gender)) {
152            Ok(key) => {
153                let data = self.banks.data(key);
154                let destination = &mut self.tracks.holocall;
155                let dilatable = true;
156                let duration: f32;
157                match data {
158                    Either::Left(data) => {
159                        duration = data.duration().as_secs_f32();
160                        if let Ok(handle) = destination.play(data) {
161                            self.tracks.store_static(
162                                handle,
163                                event_name,
164                                None,
165                                Some(emitter_name),
166                                dilatable,
167                            );
168                        }
169                    }
170                    Either::Right(data) => {
171                        duration = data.duration().as_secs_f32();
172                        if let Ok(handle) = destination.play(data) {
173                            self.tracks.store_stream(
174                                handle,
175                                event_name,
176                                None,
177                                Some(emitter_name),
178                                dilatable,
179                            );
180                        }
181                    }
182                };
183                if !emitter_name.as_str().is_empty() && emitter_name.as_str() != "None" {
184                    propagate_subtitles(
185                        event_name,
186                        EntityId::default(),
187                        emitter_name,
188                        ScnDialogLineType::Holocall,
189                        duration,
190                    );
191                } else if *key.source() == Source::Voices {
192                    warns!(
193                        "cannot propagate subtitles for voice, emitterName must be defined: {event_name}"
194                    );
195                }
196            }
197            Err(e) => {
198                warns!("cannot play over the phone: {e}");
199            }
200        }
201    }

203    pub fn play<T>(
204        &mut self,
205        event_name: CName,
206        entity_id: Option<EntityId>,
207        emitter_name: Option<CName>,
208        ext: Option<T>,
209        line_type: Option<ScnDialogLineType>,
210    ) where
211        StaticSoundData: With<Option<T>>,
212        StreamingSoundData<FromFileError>: With<Option<T>>,
213        T: AffectedByTimeDilation
214            + ValidateFor<Either<StaticSoundData, StreamingSoundData<FromFileError>>>,
215    {
216        let spoken = SpokenLocale::get();
217        let gender = entity_id.as_ref().and_then(ToGender::to_gender);
218        match self.banks.try_get(&event_name, &spoken, gender.as_ref()) {
219            Ok(key) => {
220                let data = self.banks.data(key);
221                if let Some(Err(e)) = ext.as_ref().map(|x| x.validate_for(&data)) {
222                    warns!("invalid setting(s) for audio: {e:#?}");
223                    return;
224                }
225                let duration: f32;
226                let dilatable = ext
227                    .as_ref()
228                    .map(AffectedByTimeDilation::affected_by_time_dilation)
229                    .unwrap_or(true);
230                let is_v = self
231                    .scene
232                    .as_ref()
233                    .is_some_and(|x| Some(x.listener_id()) == entity_id);
234                let destination: &mut TrackHandle = if is_v {
235                    if key.is_vocal() {
236                        &mut self.tracks.v.vocal
237                    } else {
238                        &mut self.tracks.v.emissive
239                    }
240                } else {
241                    key.to_output_destination(&mut self.tracks)
242                };
243                match data {
244                    Either::Left(data) => {
245                        duration = data.duration().as_secs_f32();
246                        if let Ok(handle) = destination.play(data.with(ext)) {
247                            self.tracks.store_static(
248                                handle,
249                                event_name,
250                                entity_id,
251                                emitter_name,
252                                dilatable,
253                            );
254                        }
255                    }
256                    Either::Right(data) => {
257                        duration = data.duration().as_secs_f32();
258                        if let Ok(handle) = destination.play(data.with(ext)) {
259                            self.tracks.store_stream(
260                                handle,
261                                event_name,
262                                entity_id,
263                                emitter_name,
264                                dilatable,
265                            );
266                        }
267                    }
268                }
269                if let (Some(entity_id), Some(emitter_name)) = (entity_id, emitter_name) {
270                    propagate_subtitles(
271                        event_name,
272                        entity_id,
273                        emitter_name,
274                        line_type.unwrap_or_default(),
275                        duration,
276                    )
277                } else if *key.source() == Source::Voices {
278                    warns!(
279                        "cannot propagate subtitles for voice, both entityID and emitterName must be defined: {event_name}"
280                    );
281                }
282            }
283            Err(e) => {
284                warns!("cannot play sound: {e}");
285            }
286        }
287    }

289    pub fn play_on_emitter<T>(
290        &mut self,
291        sound_name: CName,
292        entity_id: EntityId,
293        tag_name: CName,
294        ext: Option<T>,
295    ) where
296        StaticSoundData: With<Option<T>>,
297        StreamingSoundData<FromFileError>: With<Option<T>>,
298        T: AffectedByTimeDilation
299            + ValidateFor<Either<StaticSoundData, StreamingSoundData<FromFileError>>>,
300    {
301        if !entity_id.is_defined() {
302            warns!("cannot play sound on undefined entity: {sound_name}");
303            return;
304        }
305        let gender = entity_id.to_gender();
306        let spoken = SpokenLocale::get();
307        if let Some(ref mut scene) = self.scene {
308            match self.banks.try_get(&sound_name, &spoken, gender.as_ref()) {
309                Ok(key) => {
310                    match scene.emitters.play_on_emitter(
311                        key,
312                        &self.banks,
313                        sound_name,
314                        entity_id,
315                        tag_name,
316                        ext,
317                    ) {
318                        Err(e) => {
319                            warns!("cannot play sound on emitter: {e}");
320                        }
321                        Ok((duration, emitter_name)) => {
322                            let emitter_name = match emitter_name {
323                                Some(emitter_name) => Some(emitter_name.as_str().to_string()),
324                                None => {
325                                    let go = GameInstance::find_entity_by_id(
326                                        GameInstance::new(),
327                                        entity_id,
328                                    )
329                                    .cast::<GameObject>();
330
331                                    go.and_then(|x| {
332                                        x.is_null().not().then(|| x.resolve_display_name())
333                                    })
334                                }
335                            };
336                            if let Some(emitter_name) = emitter_name {
337                                propagate_subtitles(
338                                    sound_name,
339                                    entity_id,
340                                    CName::new(emitter_name.as_str()),
341                                    ScnDialogLineType::default(),
342                                    duration,
343                                );
344                            } else if *key.source() == Source::Voices {
345                                warns!(
346                                    "cannot propagate subtitles for voice, couldn't resolve emitter name: {sound_name} [{entity_id}]"
347                                );
348                            }
349                        }
350                    };
351                }
352                Err(e) => {
353                    warns!("cannot play sound: {e}");
354                }
355            }
356        }
357    }

374    pub fn switch<T>(
375        &mut self,
376        switch_name: CName,
377        switch_value: CName,
378        entity_id: Option<EntityId>,
379        emitter_name: Option<CName>,
380        switch_name_tween: Option<Tween>,
381        switch_value_settings: Option<T>,
382    ) where
383        StaticSoundData: With<Option<T>>,
384        StreamingSoundData<FromFileError>: With<Option<T>>,
385        T: AffectedByTimeDilation
386            + ValidateFor<Either<StaticSoundData, StreamingSoundData<FromFileError>>>,
387    {
388        if Self::exists(&switch_name) {
389            self.tracks.stop_by(
390                switch_name,
391                entity_id,
392                emitter_name,
393                switch_name_tween.unwrap_or(IMMEDIATELY),
394            );
395        } else {
396            GameInstance::get_audio_system().stop(
397                switch_name,
398                entity_id.map(Opt::from).unwrap_or(Opt::Default),
399                emitter_name.map(Opt::from).unwrap_or(Opt::Default),
400            );
401        }
402        if Self::exists(&switch_value) {
403            self.play(
404                switch_value,
405                entity_id,
406                emitter_name,
407                switch_value_settings,
408                None,
409            );
410        } else {
411            GameInstance::get_audio_system().play(
412                switch_value,
413                entity_id.map(Opt::from).unwrap_or(Opt::Default),
414                emitter_name.map(Opt::from).unwrap_or(Opt::Default),
415            );
416        }
417    }

585    pub fn set_volume(&mut self, setting: CName, value: Amplitude) {
586        lifecycle!("about to change {value} for {setting}");
587        // kira uses amplitude for volume, default to 1.
588        // while default volume expressed as game setting is 100.
589        let v = value.div(100.).clamp(0., 1.) as f64;
590        let d = crate::engine::modulators::VOLUME_MAPPING.map(v);
591        match setting.as_str() {
592            "MasterVolume" => {
593                self.manager.main_track().set_volume(d, DEFAULT);
594                self.last_volume = Some(d);
595            }
596            "SfxVolume" => self.tracks.sfx.set_volume(d, DEFAULT),
597            "DialogueVolume" => self.tracks.dialogue.set_volume(d, DEFAULT),
598            "MusicVolume" => self.tracks.music.set_volume(d, DEFAULT),
599            "CarRadioVolume" => self.tracks.car_radio.set_volume(d, DEFAULT),
600            "RadioportVolume" => self.tracks.radioport.set_volume(d, DEFAULT),
601            _ => lifecycle!("unknown volume setting: {}", setting.as_str()),
602        }
603    }

667impl ToOutputDestination for Id {
668    #[inline(always)]
669    fn to_output_destination<'b>(&self, tracks: &'b mut Tracks) -> &'b mut TrackHandle {
670        match self {
671            Id::OnDemand(_, source) | Id::InMemory(_, source) => match source {
672                Source::Sfx | Source::Ono => &mut tracks.sfx,
673                Source::Voices => &mut tracks.dialogue,
674                Source::Playlist => &mut tracks.radioport,
675                Source::Music => &mut tracks.music,
676                Source::Jingle => &mut tracks.car_radio,
677            },
678        }
