34pub struct Handle<T> {
35    pub event_name: CName,
36    pub entity_id: Option<EntityId>,
37    pub emitter_name: Option<CName>,
38    pub handle: T,
39    pub affected_by_time_dilation: bool,
40}

42#[derive(Default)]
43pub struct Handles {
44    pub statics: Vec<Handle<StaticSoundHandle>>,
45    pub streams: Vec<Handle<StreamingSoundHandle<FromFileError>>>,
46}
47
48impl Drop for Handles {
49    fn drop(&mut self) {
50        // bug in kira DecodeScheduler NextStep::Wait
51        self.streams.iter_mut().for_each(|x| {
52            x.handle.stop(IMMEDIATELY);
53        });
54    }
55}

72impl Tracks {
73    pub fn try_new<B: Backend>(
74        manager: &mut AudioManager<B>,
75        modulators: &Modulators,
76    ) -> Result<Self, Error> {
77        let ambience = Ambience::try_new(manager, modulators)?;
78        let v = V::try_new(manager, &ambience)?;
79        let holocall = Holocall::try_new(manager, &ambience)?;
80        let sfx = Sfx::try_new(manager, &ambience)?;
81        let radioport = Radioport::try_new(manager, &ambience)?;
82        let music = Music::try_new(manager, &ambience)?;
83        let dialogue = Dialogue::try_new(manager, &ambience)?;
84        let car_radio = CarRadio::try_new(manager, &ambience)?;
85        Ok(Self {
86            ambience,
87            v,
88            holocall,
89            sfx,
90            radioport,
91            music,
92            dialogue,
93            car_radio,
94            handles: Default::default(),
95        })
96    }
97    pub fn pause(&mut self, tween: Tween) {
98        self.handles.statics.iter_mut().for_each(|x| {
99            x.handle.pause(tween);
100        });
101        self.handles.streams.iter_mut().for_each(|x| {
102            x.handle.pause(tween);
103        });
104    }
105    pub fn resume(&mut self, tween: Tween) {
106        self.handles.statics.iter_mut().for_each(|x| {
107            x.handle.resume(tween);
108        });
109        self.handles.streams.iter_mut().for_each(|x| {
110            x.handle.resume(tween);
111        });
112    }
113    pub fn reclaim(&mut self) {
114        self.handles
115            .statics
116            .retain(|x| x.handle.state() != PlaybackState::Stopped);
117        self.handles
118            .streams
119            .retain(|x| x.handle.state() != PlaybackState::Stopped);
120    }
121    pub fn store_static(
122        &mut self,
123        handle: StaticSoundHandle,
124        event_name: CName,
125        entity_id: Option<EntityId>,
126        emitter_name: Option<CName>,

130            event_name,

138        &mut self,

140        event_name: CName,
141        entity_id: Option<EntityId>,
142        emitter_name: Option<CName>,

145        self.handles.streams.push(Handle {

148            emitter_name,
149            handle,
150            affected_by_time_dilation,
151        });
152    }
153    pub fn any_handle(&self) -> bool {
154        !self.handles.statics.is_empty() || !self.handles.streams.is_empty()
155    }
156    pub fn stop(&mut self, tween: Tween) {
157        self.handles.statics.iter_mut().for_each(|x| {
158            x.handle.stop(tween);
159        });
160        self.handles.streams.iter_mut().for_each(|x| {
161            x.handle.stop(tween);
162        });
163    }
164    pub fn stop_by(
165        &mut self,
166        event_name: CName,
167        entity_id: Option<EntityId>,
168        emitter_name: Option<CName>,
169        tween: Tween,
170    ) {
171        self.handles
172            .statics
173            .iter_mut()
174            .filter(|x| {
175                x.event_name == event_name
176                    && x.entity_id == entity_id
177                    && x.emitter_name == emitter_name
178            })
179            .for_each(|x| {
180                x.handle.stop(tween);
181            });
182        self.handles
183            .streams
184            .iter_mut()
185            .filter(|x| {
186                x.event_name == event_name
187                    && x.entity_id == entity_id
188                    && x.emitter_name == emitter_name
189            })
190            .for_each(|x| {
191                x.handle.stop(tween);
192            });
193    }
194    pub fn sync_dilation(&mut self, entity_id: EntityId, update: DilationUpdate) {
195        self.handles
196            .statics
197            .iter_mut()
198            .filter(|x| x.entity_id == Some(entity_id) && x.affected_by_time_dilation)
199            .for_each(|x| {
200                x.handle
201                    .set_playback_rate(update.dilation(), update.tween_curve());
202            });
203        self.handles
204            .streams
205            .iter_mut()
206            .filter(|x| x.entity_id == Some(entity_id) && x.affected_by_time_dilation)
207            .for_each(|x| {
208                x.handle
209                    .set_playback_rate(update.dilation(), update.tween_curve());
210            });
211    }
212    pub fn clear(&mut self) {
213        self.stop(IMMEDIATELY);
214        self.handles.statics.clear();
215        self.handles.streams.clear();
216    }
