11pub struct Emitter;

13impl Emitter {
14    pub fn infos(entity_id: EntityId) -> Result<(Vector4, bool), Error> {
15        let game = GameInstance::new();
16        let entity = GameInstance::find_entity_by_id(game, entity_id);
17        if entity.is_null() {
18            return Err(Error::Scene {
19                source: SceneError::MissingEmitter { entity_id },
20            });
21        }
22        let busy = entity
23            .clone()
24            .cast::<GameObject>()
25            .map(AIActionHelper::is_in_workspot)
26            .unwrap_or(false);
27        let position = entity.get_world_position();
28        Ok((position, busy))
29    }
30
31    #[allow(clippy::type_complexity)]
32    pub fn full_infos(
33        entity_id: EntityId,
34    ) -> Result<(Vector4, bool, Option<f32>, Option<SpatialTrackDistances>), Error> {
35        let (position, busy) = Self::infos(entity_id)?;
36        let game = GameInstance::new();
37        let entity = GameInstance::find_entity_by_id(game, entity_id);
38        if entity.is_null() {
39            return Err(Error::Scene {
40                source: SceneError::MissingEmitter { entity_id },
41            });
42        }
43        let distances = entity.get_emitter_distances();
44        if !entity.is_a::<TimeDilatable>() {
45            return Ok((position, busy, None, distances));
46        }
47        let dilation = entity
48            .clone()
49            .cast::<TimeDilatable>()
50            .as_ref()
51            .map(AsTimeDilatable::get_time_dilation_value);
52        Ok((position, busy, dilation, distances))
53    }
54}
