27/// Ensure no duplicate mod folder name across depots: `r6\audioware` and `mods`.
28#[inline]
29pub fn ensure_no_duplicate_accross_depots(
30    redmod_exists: bool,
31    r#mod: &Mod,
32    mods: &[Mod],
33) -> Result<(), Error> {
34    ensure!(
35        !redmod_exists || !mods.iter().any(|x| x.same_folder_name(r#mod.as_ref())),
36        DuplicateAcrossDepotsSnafu {
37            folder: r#mod.as_ref().display().to_string()
38        }
39    );
40    Ok(())

50#[cfg(feature = "hot-reload")]
51#[inline]
52pub fn ensure_key_unique_or_inserted(cname: &str) -> Result<bool, Error> {
53    // if it already existed in a previous load, it's probably a recycled key.
54    if crate::PREVIOUS_IDS
55        .lock()
56        .unwrap()
57        .iter()
58        .any(|x| AsRef::<CName>::as_ref(x).as_str() == cname)
59    {
60        return Ok(true);
61    }
62    ensure_key_unique(cname)?;
63    Ok(false)
64}

97/// Ensure [CName] does not already exist in [game pool](CNamePool),
98/// unless it already exists for some [locale](Locale).
99#[inline]
100pub fn ensure_localized_key_unique(
101    ids: &HashSet<Id>,
102    cname: &str,
103    _: Locale,
104) -> Result<bool, Error> {
105    // Conflict trait will make sure there's no identical duplicate, so no need to check twice.
106    if ids
107        .iter()
108        .any(|x| AsRef::<CName>::as_ref(x).as_str() == cname && x.locale().is_some())
109    {
110        return Ok(true);
111    }
112    ensure_key_unique(cname)?;
113    Ok(false)
114}

119    ensure!(

121        NonUniqueKeySnafu {
122            cname: cname.to_string()
123        }
124    );
125    Ok(())

144/// Ensure [Manifest] does not contain duplicate keys among
145/// [Sfx],
146/// [Ono],
147/// etc.
148pub fn ensure_manifest_no_duplicates(manifest: &Manifest) -> Result<(), Error> {
149    let mut hashset = HashSet::with_capacity(100);
150    if let Some(sfx) = manifest.sfx.as_ref() {
151        for key in sfx.keys() {
152            ensure!(
153                hashset.insert(key.as_str()),
154                ConflictingKeySnafu { cname: key.clone() }
155            );
156        }
157    }
158    if let Some(onos) = manifest.onos.as_ref() {
159        for key in onos.keys() {
160            ensure!(
161                hashset.insert(key.as_str()),
162                ConflictingKeySnafu { cname: key.clone() }
163            );
164        }
165    }
166    if let Some(voices) = manifest.voices.as_ref() {
167        for key in voices.keys() {
168            ensure!(
169                hashset.insert(key.as_str()),
170                ConflictingKeySnafu { cname: key.clone() }
171            );
172        }
173    }
174    if let Some(music) = manifest.music.as_ref() {
175        for key in music.keys() {
176            ensure!(
177                hashset.insert(key.as_str()),
178                ConflictingKeySnafu { cname: key.clone() }
179            );
180        }
181    }
182    Ok(())
183}

185/// Ensure audio file [Path] is located inside [Mod] depot.
186pub fn ensure_located_in_depot(
187    file: &impl AsRef<std::path::Path>,
188    folder: &Mod,
189) -> Result<(), Error> {
190    let arg = std::fs::canonicalize(folder)?;
191    let path = arg.join(file);
192    if !path.starts_with(arg) {
193        return Err(Error::Validation {
194            source: validation::Error::AudioOutsideDepot {
195                path: path.display().to_string(),
196            },
197        });
198    }
199    Ok(())
200}

202/// Ensure path refers to valid audio (based on [usage](Usage)).
203pub fn ensure_valid_audio(
204    path: &impl AsRef<std::path::Path>,
205    m: &Mod,
206    usage: Usage,
207    settings: Option<&Settings>,
208    captions: Option<&[Caption]>,
209) -> Result<Either<StaticSoundData, StreamingSoundData<FromFileError>>, Error> {
210    use snafu::ResultExt;
211    let filepath = m.as_ref().join(path.as_ref());
212    ensure_located_in_depot(path, m)?;
213    let data = match usage {
214        Usage::OnDemand | Usage::InMemory => StaticSoundData::from_file(filepath)
215            .context(InvalidAudioSnafu {
216                path: path.as_ref().display().to_string(),
217            })
218            .map(Either::Left)?,
219        Usage::Streaming => StreamingSoundData::from_file(filepath)
220            .context(InvalidAudioSnafu {
221                path: path.as_ref().display().to_string(),
222            })
223            .map(Either::Right)?,
224    };
225    ensure_valid_contextual_audio_settings(&data, settings, path.as_ref())?;
226    if let Some(captions) = captions {
227        ensure_valid_jingle_captions(&data, captions)?;
228    }
229    Ok(data)
230}

232/// Ensure given settings are valid for audio.
233pub fn ensure_valid_contextual_audio_settings(
234    audio: &Either<StaticSoundData, StreamingSoundData<FromFileError>>,
235    settings: Option<&Settings>,
236    path: &Path,
237) -> Result<(), Error> {
238    let mut errors = settings.validate().err().unwrap_or_default();
239    errors.extend(settings.validate_for(audio).err().unwrap_or_default());
240    ensure!(
241        errors.is_empty(),
242        InvalidAudioSettingsSnafu {
243            which: path.display().to_string(),
244            why: errors
245        }
246    );
247    Ok(())
248}

250#[doc(hidden)]
251pub fn ensure_valid_jingle_captions(
252    audio: &Either<StaticSoundData, StreamingSoundData<FromFileError>>,
253    captions: &[Caption],
254) -> Result<(), Error> {
255    if !captions.is_empty() {
256        let duration = match audio {
257            Either::Left(x) => x.duration(),
258            Either::Right(x) => x.duration(),
259        }
260        .as_secs_f32();
261        for (idx, caption) in captions.iter().enumerate() {
262            if caption.starts >= duration {
263                return Err(Error::from(validation::Error::InvalidAudioCaption {
264                    which: "starts".to_string(),
265                    why: format!("greater than audio duration (captions[{idx}])"),
266                }));
267            }
268        }
269        let mut previous_starts = captions[0].starts;
270        for (idx, caption) in captions.iter().enumerate().skip(1) {
271            if previous_starts >= caption.starts {
272                return Err(Error::from(validation::Error::InvalidAudioCaption {
273                    which: "starts".to_string(),
274                    why: format!(
275                        "unordered sequence (captions[{idx}] and captions[{}])",
276                        idx - 1
277                    ),
278                }));
279            }
280            previous_starts = caption.starts;
281        }
282    }
283    Ok(())
284}

286/// Ensure data is properly stored.
287pub fn ensure_store_data<T: PartialEq + Eq + Hash + Clone + Into<Key>>(
288    key: T,
289    value: StaticSoundData,
290    settings: Option<Settings>,
291    path: &impl AsRef<std::path::Path>,
292    store: &mut HashMap<T, StaticSoundData>,
293) -> Result<(), Error> {
294    let value = match settings {
295        Some(settings) => value.with(settings),
296        None => value,
297    };
298    ensure!(
299        store.insert(key.clone(), value).is_none(),
300        CannotStoreDataSnafu {
301            key,
302            path: path.as_ref().display().to_string()
303        }
304    );
305    Ok(())
306}

341/// Ensure guarantees are upheld.
342#[allow(clippy::too_many_arguments)]
343fn ensure<'a, K: PartialEq + Eq + Hash + Clone + Into<Key> + Conflictual>(
344    k: &'a str,
345    key: K,
346    path: PathBuf,
347    m: &Mod,
348    usage: Usage,
349    settings: Option<Settings>,
350    set: &'a mut HashSet<Id>,
351    map: &'a mut HashMap<K, StaticSoundData>,
352    smap: &'a mut HashMap<K, Settings>,
353    source: Source,
354) -> Result<(), Error>
355where
356    HashSet<Id>: Conflict<K>,
357{
358    let data = ensure_valid_audio(&path, m, usage, settings.as_ref(), None)?.map_either_with(
359        (usage, settings.as_ref().and_then(|x| x.region.clone())),
360        |ctx, data| {
361            if let (Usage::InMemory, Some(region)) = ctx {
362                data.slice(region)
363            } else {
364                data
365            }
366        },
367        |_, data| data,
368    );
369    ensure_key_no_conflict(&key, k, set)?;
370    let id: Id = match usage {
371        Usage::InMemory => Id::InMemory(key.clone().into(), source),
372        Usage::OnDemand => Id::OnDemand(
373            crate::Usage::Static(key.clone().into(), m.as_ref().join(path.clone())),
374            source,
375        ),
376        Usage::Streaming => Id::OnDemand(
377            crate::Usage::Streaming(key.clone().into(), m.as_ref().join(path.clone())),
378            source,
379        ),
380    };
381    if usage == Usage::InMemory {
382        ensure_store_data(key, data.left().unwrap(), settings, &path, map)?;
383    } else if let Some(settings) = settings {
384        ensure_store_settings(&key, settings, smap)?;
385    }
386    ensure_store_id(id, set)?;
387    Ok(())
388}

390/// Ensure [Sfx] guarantees are upheld.
391pub fn ensure_sfx<'a>(
392    k: &'a str,
393    v: Sfx,
394    m: &Mod,
395    set: &'a mut HashSet<Id>,
396    map: &'a mut HashMap<UniqueKey, StaticSoundData>,
397    smap: &'a mut HashMap<UniqueKey, Settings>,
398) -> Result<(), Error> {
399    let existed = ensure_key_unique_or_inserted(k)?;
400    let UsableAudio {
401        audio: Audio { file, settings },
402        usage,
403    } = v.into();
404    let c_string = std::ffi::CString::new(k)?;
405    let cname = CName::new(k);
406    let key = UniqueKey(cname);
407    ensure(
408        k,
409        key,
410        file,
411        m,
412        usage.unwrap_or(Usage::InMemory),
413        settings,
414        set,
415        map,
416        smap,
417        Source::Sfx,
418    )?;
419
420    if !existed {
421        CNamePool::add_cstr(&c_string);
422    }
423    Ok(())
424}

462/// Ensure [Voice] guarantees are upheld.
463#[allow(clippy::too_many_arguments)]
464pub fn ensure_voice<'a>(
465    k: &'a str,
466    v: Voice,
467    m: &Mod,
468    set: &'a mut HashSet<Id>,
469    simple: &'a mut HashMap<LocaleKey, StaticSoundData>,
470    complex: &'a mut HashMap<BothKey, StaticSoundData>,
471    simple_subs: &'a mut HashMap<LocaleKey, DialogLine>,
472    complex_subs: &'a mut HashMap<BothKey, DialogLine>,
473    simple_settings: &'a mut HashMap<LocaleKey, Settings>,
474    complex_settings: &'a mut HashMap<BothKey, Settings>,
475) -> Result<(), Error> {
476    let mut existed = false;
477    let v: AnyVoice = v.into();
478    let c_string = std::ffi::CString::new(k)?;
479    let cname = CName::new(k);
480    let mut simple_key: LocaleKey;
481    let mut complex_key: BothKey;
482    match v {
483        Either::Left((aud, usage, subs)) => {
484            for (locale, Audio { file, settings }) in aud {
485                existed = existed || ensure_localized_key_unique_or_inserted(set, k, locale)?;
486                simple_key = LocaleKey(cname, locale);
487                if let Some(subs) = subs.as_ref().and_then(|x| x.get(&locale)) {
488                    ensure_store_subtitle::<LocaleKey>(
489                        simple_key.clone(),
490                        subs.clone(),
491                        simple_subs,
492                    )?;
493                }
494                ensure(
495                    k,
496                    simple_key,
497                    file,
498                    m,
499                    usage,
500                    settings.clone(),
501                    set,
502                    simple,
503                    simple_settings,
504                    Source::Voices,
505                )?;
506            }
507        }
508        Either::Right((aud, usage, subs)) => {
509            for (locale, genders) in aud {
510                existed = existed || ensure_localized_key_unique_or_inserted(set, k, locale)?;
511                for (gender, Audio { file, settings }) in genders {
512                    complex_key = BothKey(cname, locale, gender);
513                    if let Some(subs) = subs.as_ref().and_then(|x| x.get(&locale)) {
514                        ensure_store_subtitle::<BothKey>(
515                            complex_key.clone(),
516                            if gender == PlayerGender::Female {
517                                subs.female.clone()
518                            } else {
519                                subs.male.clone()
520                            },
521                            complex_subs,
522                        )?;
523                    }
524                    ensure(
525                        k,
526                        complex_key,
527                        file,
528                        m,
529                        usage,
530                        settings.clone(),
531                        set,
532                        complex,
533                        complex_settings,
534                        Source::Voices,
535                    )?;
536                }
537            }
538        }
539    }
540
541    if !existed {
542        CNamePool::add_cstr(&c_string);
543    }
544    Ok(())
545}

583#[doc(hidden)]
584pub fn ensure_jingles<'a>(
585    k: &'a str,
586    v: Jingle,
587    m: &Mod,
588    set: &'a mut HashSet<Id>,
589    smap: &'a mut HashMap<UniqueKey, Settings>,

594    let c_string = std::ffi::CString::new(k)?;

597    ensure_key_no_conflict(&key, k, set)?;
598    let id: Id = Id::OnDemand(
599        crate::Usage::Streaming(crate::Key::Unique(key.clone()), m.as_ref().join(file)),
600        Source::Jingle,
601    );
602    if let Some(settings) = settings {
603        ensure_store_settings::<UniqueKey>(&key, settings, smap)?;
604    }
605    ensure_store_id(id, set)?;
606
607    if !existed {
608        CNamePool::add_cstr(&c_string);
609    }
610    Ok(())
