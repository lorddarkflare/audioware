39pub struct OnceStorage<K, V>(OnceLock<HashMap<K, V>>);
40impl<K, V> Default for OnceStorage<K, V> {
41    fn default() -> Self {
42        Self::new()
43    }
44}
45
46impl<K, V> OnceStorage<K, V> {
47    pub const fn new() -> Self {
48        Self(OnceLock::new())
49    }
50    pub fn set(&self, value: HashMap<K, V>) -> Result<(), HashMap<K, V>> {
51        self.0.set(value)
52    }
53    pub fn keys(&self) -> Keys<'_, K, V> {
54        self.0.get().expect("should be initialized").keys()
55    }
56    pub fn safe_keys(&self) -> Option<Keys<'_, K, V>> {
57        self.0.get().map(|x| x.keys())
58    }
59}
60impl<K: Eq + Hash> BankData for OnceStorage<K, StaticSoundData> {

62    type Data = StaticSoundData;

76
77    fn settings(&self, key: &Self::Key) -> Option<Self::Settings> {
78        self.0.get().and_then(|x| x.get(key)).cloned()

115impl BankSettings for Banks {
116    type Key = Id;
117    type Settings = audioware_manifest::Settings;
118
119    /// Retrieves sound settings for a given [Id] if any.
120    fn settings(&self, key: &Self::Key) -> Option<Self::Settings> {
121        match key {
122            Id::OnDemand(usage, ..) => {
123                let settings = match usage {
124                    Usage::Static(key, _) | Usage::Streaming(key, _) => match key {
125                        Key::Unique(key) => self.unique_settings.get(key),
126                        Key::Gender(key) => self.gender_settings.get(key),
127                        Key::Locale(key) => self.single_settings.get(key),
128                        Key::Both(key) => self.dual_settings.get(key),
129                    },
130                };
131                settings.cloned()
132            }
133            // in-memory sound data already embed settings
134            Id::InMemory(_, _) => None,
135        }
136    }
137}

139impl BankData for Banks {
140    type Key = Id;
141    type Data = Either<StaticSoundData, StreamingSoundData<FromFileError>>;
142
143    /// Retrieves sound data for a given [Id], including settings if any.
144    fn data(&self, key: &Self::Key) -> Self::Data {
145        match key {
146            Id::OnDemand(Usage::Static(_, path), ..) => {
147                let settings = self.settings(key);
148                let data = StaticSoundData::from_file(path)
149                    .expect("static sound data has already been validated");
150                if let Some(settings) = settings {
151                    return Either::Left(data.with_settings(settings.into()));
152                }
153                Either::Left(data)
154            }
155            Id::OnDemand(Usage::Streaming(_, path), ..) => {
156                let settings = self.settings(key);
157                let data = StreamingSoundData::from_file(path)
158                    .expect("streaming sound data has already been validated");
159                if let Some(settings) = settings {
160                    return Either::Right(data.with_settings(settings.into()));
161                }
162                Either::Right(data)
163            }
164            // in-memory sound data already embed settings
165            Id::InMemory(Key::Unique(key), ..) => {
166                Either::Left(self.uniques.get(key).cloned().expect("key guarantees"))
167            }
168            Id::InMemory(Key::Gender(key), ..) => {
169                Either::Left(self.genders.get(key).cloned().expect("key guarantees"))
170            }
171            Id::InMemory(Key::Locale(key), ..) => Either::Left(
172                self.single_voices
173                    .get(key)
174                    .cloned()
175                    .expect("key guarantees"),
176            ),
177            Id::InMemory(Key::Both(key), ..) => {
178                Either::Left(self.dual_voices.get(key).cloned().expect("key guarantees"))
179            }
180        }
181    }
182}

184impl BankSubtitles for Banks {
185    type Key = Id;
186    fn subtitles(&self, locale: WrittenLocale) -> Vec<(CName, (String, String))> {
187        [
188            self.single_subs
189                .iter()
190                .filter(|x| x.0.1 == locale)
191                .map(|x| (x.0.0, (x.1.msg.clone(), x.1.msg.clone())))
192                .collect::<Vec<_>>(),
193            self.dual_subs
194                .iter()
195                .filter(|x| x.0.1 == locale)
196                .map(|x| (x.0.0, (x.1.msg.clone(), x.1.msg.clone())))
197                .collect::<Vec<_>>(),
198        ]
199        .concat()
200    }
201}

203/// Returns raw entries to create a [localization package](https://github.com/psiberx/cp2077-codeware/wiki#localization-packages).
204pub trait Package {
205    fn package(&self, locale: Locale) -> Vec<(CName, (String, String))>;
206}
207
208impl Package for HashMap<LocaleKey, DialogLine> {
209    fn package(&self, locale: Locale) -> Vec<(CName, (String, String))> {
210        let mut out = Vec::new();
211        for (k, v) in self {
212            if k.1 == locale {
213                out.push((k.0, (v.msg.clone(), v.msg.clone())));
214            }
215        }
216        out
217    }
218}
219
220impl Package for HashMap<BothKey, DialogLine> {
221    fn package(&self, locale: Locale) -> Vec<(CName, (String, String))> {
222        let mut out = Vec::new();
223        let mut female: String;
224        let mut male: String;
225        for (k, v) in self {
226            if k.1 == locale {
227                if k.2 == PlayerGender::Female {
228                    female = v.msg.clone();
229                    male = self
230                        .get(&BothKey(k.0, k.1, PlayerGender::Male))
231                        .expect("genders cannot be partially defined")
232                        .msg
233                        .clone();
234                } else {
235                    male = v.msg.clone();
236                    female = self
237                        .get(&BothKey(k.0, k.1, PlayerGender::Female))
238                        .expect("genders cannot be partially defined")
239                        .msg
240                        .clone();
241                }
242                out.push((k.0, (female, male)));
243            }
244        }
245        out
246    }
